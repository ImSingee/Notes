- ## 生成质数
	- 原理：[埃拉托斯特尼筛法](https://oi-wiki.org/math/number-theory/sieve/)
	- 时间复杂度：O(n * loglogn)
	- ```go
	  var primes = get(100000)
	  
	  func get(n int) []int { // 返回 <=n 的所有素数
	      m := make([]bool, n+1)
	      primes := make([]int, 0, n+1)
	  
	      for i := 2+1; i <= n; i++ {
	          if !m[i] { // is prime
	              primes = append(primes, i)
	              for x := 2 * i; x <= n; x += i {
	                  m[x] = true
	              }
	          }
	      }
	      
	      return primes
	  }
	  ```
