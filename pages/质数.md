- ## 生成质数
	- 原理：[埃拉托斯特尼筛法](https://oi-wiki.org/math/number-theory/sieve/)
	- 时间复杂度：O(n * loglogn)
	- ```go
	  var primes []int
	  
	  func init() {
	      N := 317 // 返回小于等于 N 的所有素数
	      m := make([]bool, N+1)
	      primes = make([]int, 0, N)
	      for i := 2; i <= N; i++ {
	          if !m[i] { // is prime
	              primes = append(primes, i)
	              for x := 2 * i; x <= N; x += i {
	                  m[x] = true
	              }
	          }
	      }
	  }
	  ```
	- ```go
	  func get(n int) []int { // 返回 <=n 的所有素数
	      m := make([]bool, n+1)
	      primes := make([]int, 0, n+1)
	  
	      for i := 2+1; i <= n; i++ {
	          if !m[i] { // is prime
	              primes = append(primes, i)
	              for x := 2 * i; x <= n; x += i {
	                  m[x] = true
	              }
	          }
	      }
	      
	      return primes
	  }
	  ```
