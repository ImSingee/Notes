- #Algorithm #Leetcode [#Leetcode-200](https://leetcode-cn.com/problems/number-of-islands/) #Medium
-
- ## 最优解法：
	-
	- ```go
	  func numIslands(grid [][]byte) int {
	      count := 0
	  
	      for i := range grid {
	          for j := range grid[i] {
	              if grid[i][j] != '1' {
	                  continue
	              }
	  
	              count++
	              mark(grid, i, j)
	          }
	      }
	  
	      return count
	  }
	  
	  func mark(grid [][]byte, i, j int) {
	      if i < 0 || i >= len(grid) || j < 0 || j >= len(grid[0]) {
	          return
	      }
	      if grid[i][j] != '1' {
	          return
	      }
	  
	      grid[i][j] = 0
	      mark(grid, i-1, j)
	      mark(grid, i+1, j)
	      mark(grid, i, j-1)
	      mark(grid, i, j+1)
	  }
	  ```
