- #+BEGIN_QUERY
  {:title "All pages have a *每日阅读* tag"
   :query [:find ?name
         :in $ ?tag
         :where
         [?t :block/name ?tag]
         [?p :page/tags ?t]
         [?p :block/name ?name]]
   :inputs ["每日阅读"]
   :view (fn [result]
         [:div.flex.flex-col
          (for [page result]
            [:a {:href (str "/page/" page)} (clojure.string/capitalize page)])])}
  #+END_QUERY
